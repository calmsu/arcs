We don't store the size the image is displayed at when hotspotted, so actual coordinates are useless. Scale x,y to the range (0,1).

 
    _scaleDown: (cds) ->
        cds.x1 /= @img.width()
        cds.y1 /= @img.height()
        cds.x2 /= @img.width()
        cds.y2 /= @img.height()
        cds

    _scaleUp: (cds) ->
        cds.x1 *= @img.width()
        cds.y1 *= @img.height()
        cds.x2 *= @img.width()
        cds.y2 *= @img.height()
        cds

    update: ->
        @collection.fetch
            success: =>
                @.render()

    render: ->
        $hotspots = $('#hotspots-wrapper')
        $annotations = $('#annotations-wrapper')
        $hotspots.html ''
        $annotations.html ''

        json = 
            hotspots: []
        for hotspot in @collection.models
            data = @._scaleUp(hotspot.toJSON())
            data.left = data.x1
            data.width = data.x2 - data.x1
            data.top = data.y1
            data.height = data.y2 - data.y1
            json.hotspots.push data

        $hotspots.html Mustache.render arcs.templates.hotspot, json
        $annotations.html Mustache.render arcs.templates.annotation, json
